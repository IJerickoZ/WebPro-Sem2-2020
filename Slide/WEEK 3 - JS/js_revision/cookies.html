JS Cookies & Local Storage

1. Creating cookies

    function createCookie(){
        var value = document.getElementById('input1').value
        // Create a cookie, valid across the entire site:
        Cookies.set('myCookie', value)
        // Create a cookie that expires 7 days from now, valid across the entire site:
        Cookies.set('myCookie7', value, { expires: 7 })

        document.getElementById('input1').innerHTML = Cookies.get('myCookie')
    }

2. Reading cookies

    function readCookie(){
        var myCookie = Cookies.get('myCookie')
        document.getElementById('result1').innerHTML = myCookie
    }

3. Deleting cookies

    function deleteCookie(){
        Cookies.remove('myCookie')
    }

4. Local Storage

The localStorage object stores data with no expiration date. The data will not be deleted when the browser is closed, and will be available the next day, week, or year. 

    function setLocalStorage(){
        var value = document.getElementById('input1').value
        localStorage.setItem("myVar", value)
    }

    function getLocalStorage(){
        var lastname = localStorage.getItem("myVar")
    }

    function removeLocalStorage(){
        localStorage.removeItem("myVar")
    }
